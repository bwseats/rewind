                        // controller function
                        function getValues() {
                            let inputString = document.getElementById('reverse').value;
                        
                            let reversedString = reverseAString(inputString);
                            displayString(reversedString);
                        }
                        
                        // logic function
                        function reverseAString(userString) {
                        
                            let revString = '';
                        
                            for(let i = userString.length - 1; i >= 0; i--) {
                                let currentLetter = userString[i];
                            
                                revString += currentLetter
                            }
                        
                            return revString;
                        }
                        
                        // view function
                        function displayString(revString) {
                            document.getElementById('results').textContent = revString;
                            document.getElementById('alert').classList.remove('invisible');
                        }

The code is structured in three functions:

getValues()

This function gets the user input string and passes it as a parameter to the next function:

reverseAString()

This function has a decrementing for loop to count from the end of the user input string and create a new string in reverse. This string is then passed to the final function:

displayString()

This function simply displays the reversed string in an alert and removes the 'invisible' class from the alert.
